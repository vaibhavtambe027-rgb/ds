# Data Visualization III - Iris Dataset

# Step 1: Import Libraries
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

# Step 2: Load Dataset
df = sns.load_dataset('iris')

# Step 3: Display Dataset
print("First 5 Rows:\n")
print(df.head())

# Step 4: Shape of Dataset
print("\nShape of Dataset:")
print(df.shape)

# Step 5: Features and Datatypes
print("\nFeatures and Their Types:\n")
print(df.dtypes)

# Step 6: Histogram for Each Feature
print("\nDisplaying Histograms...")

df.hist(figsize=(10,8))

plt.suptitle("Histogram of Iris Dataset Features")

plt.show()

# Step 7: Boxplot for Each Feature
print("\nDisplaying Boxplots...")

numeric_columns = df.select_dtypes(include=['float64', 'int64']).columns

for column in numeric_columns:

    plt.figure(figsize=(6,4))

    sns.boxplot(x=df[column])

    plt.title(f"Boxplot of {column}")

    plt.show()

# Step 8: Outlier Detection and Inference
print("\nInference:\n")

for column in numeric_columns:

    Q1 = df[column].quantile(0.25)

    Q3 = df[column].quantile(0.75)

    IQR = Q3 - Q1

    lower_bound = Q1 - 1.5 * IQR

    upper_bound = Q3 + 1.5 * IQR

    outliers = df[(df[column] < lower_bound) | (df[column] > upper_bound)]

    print(f"{column}:")

    print("Number of Outliers =", len(outliers))

    print("---------------------------")
